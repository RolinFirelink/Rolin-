#include <stdio.h>
int main()
{
	const int number = 100;
	int a[number];
	int i,l,k;
	int n=0,m=0;
	double ave;
	for(l=0;l<number;l++)
	{
		a[l]=0;
	}
	for( i=0;i<number;i++)
	{
		scanf("%d", &a[i]);
		n++;
		m+=a[i];
		if (a[i]==-1)
		{
			break;
		}
	}
	ave= m/n;
	for(k=0;k<number;k++)
	{
		if( a[k]>ave)
		{
			printf("%d\n",a[k]);
		}
	}
	return 0;
}
