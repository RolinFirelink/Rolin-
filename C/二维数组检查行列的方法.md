#include <stdio.h>
int main()
{
	const int size = 3;
	int a[size][size];
	int i,j;
	int x=0,y=0,cnt=0;
	
	//对数组进行初始化 
	for (i=0;i<size;i++)
	{
		for(j=0;j<size;j++)
		{
			a[i][j] = 0;
			scanf("%d", &a[i][j]);
		}
	}
	
	//检查横列 
	for(i=0;i<size;i++)
	{
		for(j=0;j<size;j++)
		{
			if ( a[i][j]==0 )
			{
				x++;
			}
			else
			{
				y++;
			}
			if( x==size)
			{
				cnt++;
				printf("X赢了");
				break;
			}
			if( y==size)
			{
				cnt++;
				printf("Y赢了");
				break;
			}
		}
		if ( cnt!=0)
		{
			break;
		}
	}
	
	if (cnt!=0)
	{
		return 0;
	}
	
	x=0;y=0; //x,y的初始化 
	
	//检查纵列 
	for(j=0;j<size;i++)
	{
		for(i=0;i<size;j++)
		{
			if ( a[j][i]==0 )
			{
				x++;
			}
			else
			{
				y++;
			}
			if( x==size)
			{
				cnt++;
				printf("X赢了");
				break;
			}
			if( y==size)
			{
				cnt++;
				printf("Y赢了");
				break;
			}
		}
		if ( cnt!=0)
		{
			break;
		}
	}
	
	if (cnt!=0)
	{
		return 0;
	}
	
	x=0;y=0;
	
	//检查对角线 
	for(i=0;i<size;i++)
	{
		if ( a[i][i]==0 )
			{
				x++;
			}
			else
			{
				y++;
			}
			if( x==size)
			{
				cnt++;
				printf("X赢了");
				break;
			}
			if( y==size)
			{
				cnt++;
				printf("Y赢了");
				break;
			}
		
		if ( cnt!=0)
		{
			break;
		}
	}
	
	if (cnt!=0)
	{
		return 0;
	}
	
	x=0;y=0;
	
	//检查另一边的对角线 
	for(i=0;i<size;i++)
	{
		if ( a[i][size-i+1]==0 )
			{
				x++;
			}
			else
			{
				y++;
			}
			if( x==size)
			{
				cnt++;
				printf("X赢了");
				break;
			}
			if( y==size)
			{
				cnt++;
				printf("Y赢了");
				break;
			}
		
		if ( cnt!=0)
		{
			break;
		}
	}
	
	if (cnt!=0)
	{
		return 0;
	}
	
	printf("本局无人胜利"); 
	
	
	
	return 0;
}
